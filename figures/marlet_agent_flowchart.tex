\node[red, below=9mm of score] (bonus) {Field bonus $B_e(x)$\\\footnotesize context +0.08\\image +0.05};

\draw[evidence] (bonus) -- (se);

\node[red, right=9mm of reranker] (packer) {Context packer\\\footnotesize MMR, max 4 chunks\\6000 chars};

\draw[evidence] (packer) -- (evidenceChunks);
